\section{Estabilidad en cuerda - Caso estocástico}

\begin{frame}
\frametitle{Estabilidad en cuerda estocástica}
$\qquad \circ$ Campo poco explorado. \\
\vspace{5mm}
$\qquad \circ$ Análisis de estabilidad en cuerda estocástica está restringido o ausente. \\
\vspace{5mm}
$\qquad \circ$ Definiciones propuestas aplican para la media o el segundo momento. \\
\vspace{5mm}
$\qquad \circ$ Dichas definiciones no se pueden aplicar directamente a pelotones \\ 
$\hspace{10mm}$ con canales afectados por ruido aditivo. \\
\vspace{5mm}
$\qquad \circ$ Condiciones necesarias y suficientes.
\end{frame}


\begin{frame}
\frametitle{Estabilidad en cuerda estocástica}
{\bf \large $\bullet$ Estabilidad en cuerda $\mathcal{L}_p-\mathcal{L}_q$:}\\
\vspace{5mm}
\begin{mybox}{}
{\bf Definición 3:}
Un pelotón de vehículos se dice estable en cuerda en base a la media con norma $\mathcal{L}_p$ y la varianza con norma $\mathcal{L}_q$, donde $p,q \in \mathbb{N}\cup\{+\infty\}$, si existen las siguientes funciones clase $\mathcal{K}$, $\alpha_1$, $\alpha_2$, $\beta_1$ y $\beta_2$, tales que, para cualquier $\{\bm{\xi}(0),\bm{\nu}\} \in \mathcal{G}$, se cumple
\begin{align*}
\norm{\mu_{\zeta_i}(k)}_{\mathcal{L}_p} &\leq \alpha_1(\norm{\mu_{\bm{\nu}}(k)}_{\mathcal{L}_p})+\beta_1(\norm{\mu_{\bm{\xi}}(0)}_p), \\
\norm{P_{\zeta_i}(k)}_{\mathcal{L}_q} &\leq \alpha_2 (\norm{P_{\bm{\nu}}(k)}_{\mathcal{L}_q})+\beta_2(\norm{P_{\bm{\xi}}(0)}_q), %\\
\end{align*}
 $\forall k\in\mathbb{N}_0$, $\forall i \in \left\lbrace 1, \dots, N\right\rbrace$ y $\forall N \in \mathbb{N}$.
\end{mybox}
\vspace{1mm}
\end{frame}

\begin{frame}{}

{\bf \large $\bullet$ Estabilidad en cuerda en sentido cuadrático medio} \\
\vspace{5mm}
\begin{mybox}{}
{\bf Definición 4:}
Un pelotón que satisface la Definición 3 se dice \textbf{estable en cuerda en sentido cuadrático medio} si y solo si, para cualquier $\{\bm{\xi}(0),\bm{\nu} \} \in \mathcal{G}$, existen constantes finitas $M_1$ y $M_2$ tales que
\begin{align*}
    \lim_{k,i \rightarrow \infty}\mu_{\zeta_i}(k)  = M_1, \quad \lim_{k,i \rightarrow \infty}P_{\zeta_i}(k)  = M_2. 
\end{align*}
\end{mybox}
\end{frame}

\section{Análisis del pelotón con ruido en todos los agentes}
\begin{frame}
\frametitle{Configuración del pelotón}
\begin{figure}[h!]
\includegraphics[width=10.5cm,height=3.5cm]{Figures/config_all.eps}
\end{figure}
$T(z)$ está descrita por
\begin{subequations}
    \label{eq:sys_estado_x_def}
    \begin{align}
	     x_i(k+1)&=A x_i(k) + B {y}_{i-1}(k) + B d_{i}(k)\\
	     y_i(k)&=C x_i(k)
	\end{align}
\end{subequations}
%\pause
{\bf \large $\bullet$ Estado alternativo}
\begin{equation}
    \label{eq:estado_alternativo_def}
    \xi_i(k) \triangleq x_{i-1}(k)-x_i(k)- h\left[ x_i(k)-x_i(k-1) \right]
\end{equation} 
\begin{equation}
    \label{eq:estado_alternativo_zero}
    \zeta_0(k)=e_0(k)=r_0(k)-y_0(k)- h\left[ y_0(k)-y_0(k-1) \right].
\end{equation}
\end{frame}


\begin{frame}{}
\frametitle{Representación en espacio de estados del pelotón}
\begin{subequations} \label{eq:ss_sys_all}
    \begin{align}
        \label{eq:xi_all}
        \bm{\xi}(k+1) &=\mathbf{A} \bm{\xi}(k) + \mathbf{B}_{o}\, \zeta_0(k)+\mathbf{B}_{a}\, \mathbf{d}(k) +\mathbf{B}_{b}\, \mathbf{d}(k-1) \\
        \label{eq:z_all}
        \bm{\zeta}(k) &= \mathbf{C} \bm{\xi}(k),
    \end{align}
\end{subequations}
\begin{align*}
    \mathbf{A} &\triangleq 
        \begin{bmatrix}
            A & \ & \ & \ \\
            BC & A & \ & \ \\
            \ & \ddots & \ddots & \ \\
            \ & \ & BC & A
        \end{bmatrix}, \;
    \mathbf{B}_{o} \triangleq  \begin{bmatrix}
        B \\ 0 \\ \vdots \\ 0
        \end{bmatrix}, \;
    \mathbf{B}_{a} \triangleq \begin{bmatrix}
            \lambda B &   &  \  &  \  \\
            B &  \lambda B &    \  &  \  \\
            & \ddots  &  \ddots & \ \\
            &  & B & \lambda B
        \end{bmatrix}, \\
    \mathbf{d}(k) &\triangleq  \begin{bmatrix}
        d_1(k) \\ d_2(k) \\ \vdots \\ d_N(k) 
        \end{bmatrix}, \;
    \mathbf{B}_{b} \triangleq \begin{bmatrix}
            h B &   &  \  &  \  \\
             &  h B &    \  &  \  \\
            &   &  \ddots & \ \\
            &  &  & h B
        \end{bmatrix}, \;
    \mathbf{C} \triangleq \begin{bmatrix}
           C &   &  \  &  \  \\
             &  C &    \  &  \  \\
             &   &  \ddots & \ \\
             &  &  & C
        \end{bmatrix},
\end{align*}
$\hspace{14mm}$ con $\lambda\triangleq -(1+h)$.
\end{frame}

\begin{frame}{}
\frametitle{Estadísticas del estado y el error de seguimiento}
\vspace{3mm}
\begin{mybox}{}
{\bf Lema 1:} Las medias del estado y el error de seguimiento están dadas por 
\begin{subequations} \label{eq:means_all}
\begin{align}
    \label{eq:mu_xi_all}
    \mu_{\bm{\xi}}(k+1) &= \mathbf{A} \mu_{\bm{\xi}}(k) +\mathbf{B}_o\ \zeta_0(k), \quad \mu_{\bm{\xi}}(0)=\mu_{\bm{\xi}_0}, \\
        \label{eq:mu_z_all}
    \mu_{\bm{\zeta}}(k) &=
        \mathbf{C} \mu_{\bm{\xi}}(k)
\end{align}	  
\end{subequations}
y sus matrices de varianza son
\begin{subequations} \label{eq:variance_all}
\begin{align} 
    \label{eq:P_xi_all}
    P_{\bm{\xi}}(k+1) &=
    \begin{cases}
        \mathbf{A} P_{\bm{\xi}}(k)\mathbf{A}^\top + \mathbf{B}_a P_{\bm{d}} {\mathbf{B}_a}^\top,  & \text{for } k=0, \\
        \mathbf{A} P_{\bm{\xi}}(k)\mathbf{A}^\top + \Upsilon, & \text{for } k>0,
    \end{cases} \\
    \label{eq:P_z_all}
    P_{\bm{\zeta}}(k) &=
        \mathbf{C} P_{\bm{\xi}}(k)\mathbf{C}^\top,
\end{align}
\end{subequations}
donde $P_{\bm{\xi}}(0) = P_{\bm{\xi}_0} \geq 0$, $\Upsilon = \mathbf{B}_{a} P_{\mathbf{d}} {\mathbf{B}_{a}}^\top + \mathbf{B}_{b} P_{\mathbf{d}} {\mathbf{B}_{b}}^\top + \mathbf{A}\mathbf{B}_{a} P_{\mathbf{d}} {\mathbf{B}_{b}}^\top + \mathbf{B}_{b} P_{\mathbf{d}} {\mathbf{B}_{a}}^\top \mathbf{A}^{\top}$ y $P_{\bm{d}}=diag(P_d,P_d,\cdots,P_d)$. 
\end{mybox}
\end{frame}


\begin{frame}{}
\frametitle{Condición de convergencia en el tiempo}
\vspace{3mm}
\begin{mybox}{}
    {\bf Lema 2:} La media y varianza del error de seguimiento, $\mu_{\bm{\zeta}}(k)$ y $P_{\bm{\zeta}}(k)$ convergen (en el tiempo) a valores estacionarios, para todo $N \in \mathbb{N}$, si y solo si
    \begin{align}
        \label{eq:MSS_all}
        \rho (A)<1.
    \end{align}
   Dichos valores estacionarios están dados por
   \begin{subequations} \label{eq:stationary_all}
       \begin{align}
           \mu_{\bm{\zeta}} =& 0 \\
           P_{\bm{\zeta}} =& vec^{-1} \left( (\mathbf{C} \otimes \mathbf{C})  (I- \mathbf{A} \otimes \mathbf{A} )^{-1} vec(\Upsilon)  \right).
       \end{align}
   \end{subequations}
\end{mybox}
\end{frame}

\begin{frame}{}
\frametitle{Escalabilidad: Condición de convergencia en el número de vehículos}
\vspace{3mm}
\begin{mybox}{}
    {\bf Teorema 1:} Considere el sistema descrito en \eqref{eq:ss_sys_all} y asumiendo que $\rho(A) < 1$. La media y la densidad espectral de potencia del error de seguimiento del $i$-ésimo vehículo, con $i \in \lbrace 2,3,...,N\rbrace$, $\forall N \in \mathbb{N}$, satisfacen
    \begin{align}
    \label{eq:zdom_muz_all}
    \mu_{\zeta_{i}}(e^{j\omega}) =& T(e^{j\omega}) \mu_{\zeta_{i-1}} (e^{j\omega}) \\
     \label{eq:zdom_psdz_all}
    \phi_{\zeta_i}(e^{j\omega}) =&
        |T(e^{j\omega})|^2 \phi_{\zeta_{i-1}} (e^{j\omega}) + |T(e^{j\omega})|^2 \ |S(e^{j\omega})|^2 P_d + \left( 1-|T(e^{j\omega} \right)|^2)\ |1-S(e^{j\omega})|^2  P_d,
    \end{align}
    donde $\mu_{\zeta_{1}}(e^{j\omega}) = T(e^{j\omega}) \zeta_{0} (e^{j\omega})$, y $\phi_{\zeta_1}(e^{j\omega})=|H(e^{j\omega})T(e^{j\omega})|^2 P_d$.
    Adicionalmente, $\mu_{\zeta_{N}}(e^{j\omega})$ y $\phi_{\zeta_N}(e^{j\omega})$ permanecen acotados cuando $N \rightarrow \infty$, $\forall \omega$, si y solo si 
        \begin{equation} \label{eq:scalability_all}
    |T(e^{j\omega})| <1, \quad \forall \omega >0.
     %\norm{T(z)}_\infty \leq 1
\end{equation}
\end{mybox}
\end{frame}

\begin{frame}{}
\frametitle{Escalabilidad: Condición de convergencia en el número de vehículos}
\vspace{3mm}
\begin{mybox}{}
    {\bf Corolario 1:} Considere el pelotón descrito en \eqref{eq:ss_sys_all} y que este satisface las condiciones \eqref{eq:MSS_all} y \eqref{eq:scalability_all}. Entonces, para $i \in \lbrace 1,2,...,N\rbrace$, $\forall N \in \mathbb{N}$, las varianzas estacionarias $P_{\zeta_i}$ satisfacen
\begin{flalign*}
    &1. \quad P_{\zeta_{i-1}} \leq  P_{\zeta_i} \qquad \forall i>1 & \\
    &2. \quad \max_{i} P_{\zeta_i} =  P_{\zeta_N}  =\|F_{\zeta_N}\|_{2}^2 \; P_d  &  \\
    &3. \quad \lim_{N \rightarrow \infty} P_{\zeta_N}=\left( \norm{ \dfrac{S(z)}{M(z)} }_2^2 -1 \right) P_d.&
\end{flalign*}
donde $F_{\zeta_N} = \left[ ST^{N-1}\ \ ST^{N-2}\ \ \cdots \  ST \, \, HT \right]$ y $M(z)$ es un factor espectral de fase mínima y estable, tal que $1-T(z)T(z)^{\sim}=M(z)M(z)^{\sim}$.
\end{mybox}
\end{frame}


\begin{frame}{}
\frametitle{Análisis de estabilidad en cuerda}
\vspace{3mm}
\begin{mybox}{}
    {\bf Teorema 2:} Considere el pelotón descrito en \eqref{eq:ss_sys_all}. Las condiciones en \eqref{eq:MSS_all} y \eqref{eq:scalability_all} son necesarias y suficientes para lograr estabilidad en cuerda  en base a la media con norma $\mathcal{L}_2$ y la varianza con norma $\mathcal{L}_\infty$. Además, dichas condiciones también permiten lograr estabilidad en cuerda en sentido cuadrático medio. Los valores límite para la media y la varianza del error de seguimiento están dados por
\begin{align*}
\lim_{k,N \rightarrow \infty}\mu_{\zeta_N}(k) & = 0 ,\\
\lim_{k,N \rightarrow \infty}P_{\zeta_N}(k)  &= \left( \norm{ \frac{S(z)}{M(z)} }_2^2 -1 \right) P_d .
\end{align*}
\end{mybox}
\end{frame}

\begin{frame}

\begin{figure}[h!]

\begin{subequations} \label{eq:ss_sys_lider}
    \begin{align}
        \label{eq:xi_lider}
        \bm{\xi}(k+1) &=\mathbf{A} \bm{\xi}(k) + \mathbf{B}_{o}\, \zeta_0(k)+\mathbf{B}_{a}\, d_1(k) +\mathbf{B}_{b}\, d_1(k-1) \\

&1.\quad P_{\zeta_i} \leq  P_{\zeta_{i-1}} \quad \forall 
 i>1 & \\
 &2. \quad \max_{i} P_{\zeta_i} =  P_{\zeta_1}  =\| H(z)T(z) \|_{2}^2 P_{d_1} & \\
 &3. \quad \lim_{N \rightarrow \infty} P_{\zeta_N}=0. &

\begin{frame}{}
\frametitle{Análisis de estabilidad en cuerda}
\vspace{3mm}
\begin{mybox}{}
    {\bf Teorema 3:} Considere el sistema descrito en \eqref{eq:ss_sys_lider}. Las condiciones $\rho(A) < 1$ y $\norm{T(z)}_\infty \leq 1$ son necesarias y suficientes para conseguir estabilidad en cuerda en base a la media con norma $\mathcal{L}_2$ y la varianza con norma $\mathcal{L}_\infty$. Además, dichas condiciones también permiten conseguir estabilidad en cuerda en sentido cuadrático medio. Los valores límite de la media y la varianza del error de seguimiento están dados por
\begin{align*}
\lim_{k,N \rightarrow \infty}\mu_{\zeta_N}(k) & = 0 ,\\
\lim_{k,N \rightarrow \infty}P_{\zeta_N}(k) & = 0 .
\end{align*}
\end{mybox}
\end{frame}

\begin{frame}{Estabilidad de Lyapunov}
Sea el sistema LTI
\begin{equation*}
    x(t+1)=A x(t), \qquad A \in \mathcal{R}^{n\times n}
\end{equation*}
con un punto de equilibrio en el origen $x=0$. La solución del sistema LTI tiene la forma
\begin{equation*}
    x(t)=A^t x(0)
\end{equation*}
Entonces, el punto de equilibrio es estable si y solo si todos los valores propios de $A$ satisfacen $|\lambda_i|\leq 1$. Una matriz $A$ con todos los valores propios (en valor absoluto) menores que $1$ se denomina matriz {\bf Schur}.\\

{\it Teorema:} Un sistema de tiempo discreto con la matriz del sistema $A$ (Schur) es asintóticamente estable si y solo si existe una única matriz $P$ que satisface la ecuación de Lyapunov
\begin{equation*}
    A^{\top}PA-P=-Q
\end{equation*}
para toda matriz definida positiva $Q$.
\end{frame}
